Table customers {
  customer_id int [pk, increment]
  name varchar(150)
  email varchar(150)
  phone varchar(15)
}

Table rental_records {
  rental_id int [pk, increment]
  tractor_id int
  customer_id int
  start_date date
  end_date date
}

Table tractors {
  tractor_id int [pk, increment]
  model varchar(100)
  brand varchar(100)
  type varchar(100)
  availability tinyint(1)
}

Ref: rental_records.tractor_id > tractors.tractor_id
Ref: rental_records.customer_id > customers.customer_id
